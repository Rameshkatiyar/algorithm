public class QuickSort {

    public static void main(String[] args) {

    }

    public static void quickSort(int[] a, int start, int end){
        if(start < end){
            int pIndex = partion(a, start, end);
            quickSort(a, start, pIndex-1);
            quickSort(a, pIndex+1, end);
        }
    }

    public static void partion(int[] a, int start, int end){
        int pivot = a[end];
        int pIndex = 0;
        for(int i=0; i<=end; i++){
            if(a[i] < pivot){
                swap(a, i, pIndex);
                pIndex++;
            }
        }
        swap(a, pIndex, end);
        return pIndex;
    }

    public static void swap(int[] a, i, j){
        int temp = a[i];
        a[i] = a[j];
        a[j] = temp;
    }
}
